class Solution {
public:
    bool isPowerOfThree(int n) {
        if(n<=0) return false;

        while(n%3==0){
            n=n/3;
        }
        if(n==1) return true;
        return false;
    }
};

// Time Complexity : O(log3(n));



// Optimal solution
class Solution {
public:
    bool isPowerOfThree(int n) {
        // 1. A power of three must be positive.
        if (n <= 0) {
            return false;
        }
        
        // 2. 1162261467 is the largest power of 3 that fits in a 32-bit signed integer (3^19).
        // If n is a power of three, it must be a divisor of this number.
        return 1162261467 % n == 0;
    }
};

class Bank {
private:
    vector<long long> accBalances;
    int n;

public:
    Bank(vector<long long>& balance) {
        accBalances = balance;
        n = accBalances.size();
    }
    
    bool transfer(int account1, int account2, long long money) {
        if (account1 < 1 || account1 > n || account2 < 1 || account2 > n) {
            return false;
        }
        
        if (accBalances[account1 - 1] < money) {
            return false;
        }
        
        accBalances[account1 - 1] -= money;
        accBalances[account2 - 1] += money;
        return true;
    }
    
    bool deposit(int account, long long money) {
        if (account < 1 || account > n) {
            return false;
        }
        
        accBalances[account - 1] += money;
        return true;
    }
    
    bool withdraw(int account, long long money) {
        if (account < 1 || account > n) {
            return false;
        }
        
        if (accBalances[account - 1] < money) {
            return false;
        }
        
        accBalances[account - 1] -= money;
        return true;
    }
};

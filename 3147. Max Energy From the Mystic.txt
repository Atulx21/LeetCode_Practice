class Solution {
public:
    int maximumEnergy(vector<int>& energy, int k) {
        int n = energy.size();
        vector<long long> dp(n, 0);
        long long ans = LLONG_MIN;

        for (int i = n - 1; i >= 0; --i) {
            long long next = (i + k < n) ? dp[i + k] : 0;
            dp[i] = energy[i] + next;
            ans = max(ans, dp[i]);
        }
        
        return ans;
    }
};

class Solution {
  public:
    long long solve(vector<int>& bt) {
        // code here
        sort(bt.begin(),bt.end());
         int temp=0;
        int sum=0;
        for(int i=0;i<bt.size()-1;i++){
             temp=temp+bt[i];
            sum += temp;
        }
        return (sum/bt.size());
        
        
    }
};
